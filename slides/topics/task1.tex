    \begin{block}{Hint!}
        To print to the terminal, use:
        
        \mint{dart}| import 'dart:developer' as developer;|

        \mint{dart}| developer.log('Hello!', name: 'app.log');|
    \end{block}
\end{frame}

\end{frame}
